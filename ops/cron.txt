# Cron jobs for ops/telegram_ops.py
#
# Install with:
#   crontab -e
# then paste these lines. Replace REPO_DIR if the bot is checked out elsewhere.
#
# Required environment (set in crontab, NOT in .bashrc — cron has a minimal env):
#   TELEGRAM_BOT_TOKEN=...
#   TELEGRAM_CHAT_ID=...
#   DREAMLEAGUE_ACTIVE=true        # only triggers liveness alerts when set
#
# Tip: set them in the crontab header lines so all jobs inherit:
#
#   TELEGRAM_BOT_TOKEN=...
#   TELEGRAM_CHAT_ID=...
#   DREAMLEAGUE_ACTIVE=true
#   REPO_DIR=/home/tstuv/dota/dotesbot-main/dotesbot-main

# Liveness check every 30 minutes. Script no-ops when window is inactive or the
# most recent live_attempts.csv row is within --max-idle-hours.
*/30 * * * * cd $REPO_DIR && .venv/bin/python3 ops/telegram_ops.py liveness --max-idle-hours 6 >> logs/cron_telegram.log 2>&1

# Daily summary at 09:00 UTC.
0 9 * * * cd $REPO_DIR && .venv/bin/python3 ops/telegram_ops.py daily --hours 24 --require-send >> logs/cron_telegram.log 2>&1
